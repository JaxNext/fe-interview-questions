  function func(arr) {
    const res = []
    const stack = [...arr]

    while (stack.length > 0) {
      const current = stack.pop()
      if (!Array.isArray(current)) {
        res.unshift(current)
      } else {
        stack.push(...current)
      }
    }
    return res
  }
  console.log(func([1, [5,4], [[[3]]], 0]))
