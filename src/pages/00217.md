  function func(arr, target) {
    let start = 0, end = arr.length - 1
    
    while(start <= end){
      const midIndex = Math.floor((start + end) / 2)
      const x = arr[midIndex]
      if (target === x) {
        return midIndex
      } else if (target < x) {
        end = midIndex - 1
      } else {
        start = midIndex + 1
      }
    }
    return -1
  }
  console.log(func([1, 3, 5, 7, 9, 11, 13], 5))
