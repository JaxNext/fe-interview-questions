# 总结



<script>
  function func(num) {
    return num.toString().split('').sort((a, b) => b - a)
  }

  console.log(func(123))
  console.log(func(321))
  console.log(func(100))
  console.log(func(101))
  console.log(func(2349758347348))
  console.log(func(111))
  
  
</script>
