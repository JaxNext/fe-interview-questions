# Q

给定一个字符串，返回其中最短的词的长度

# A



# 总结



<script>
  function func(str) {
    const arr = str.split(' ')
    let minLen = arr[0].length
    for (const x of arr) {
      if (x.length < minLen) {
        minLen = x.length
      }
    }
    return minLen
  }
  console.log(func('The quick brown fox jumps over the lazy dog')) // 3
  console.log(func('Hello world')) // 5
  console.log(func('JavaScript is fun')) // 2
  
  
</script>
