# 题干

- 给定一个数组
- 给定一个最大值
- 给定加数的个数
- 求不大于最大值的和
- 若无，返回 null

# 总结



<script>
  function func(arr, max, count) {
    if (!arr
      || arr.length < count
      || count < 2
    ) return null

    let sum = 0
    for (let i = 0; i < arr.length; i++) {
      if (sum + arr[i] > max
        || count < 2
      ) return sum
      sum += arr[i]
      count--
    }

    return sum
  }
  console.log(func([1, 2, 3, 4, 5], 10, 9))
</script>
