# Q

- 给定一个数组，长度为奇数，元素为整数；
- 数组中只有一个元素与其他不同，找到它。

# A



# 总结



<script>
  function func(arr) {
    const map = new Map()
    for (const x of arr) {
      map.set(x, (map.get(x) || 0) + 1)
    }
    for (const [k, v] of map.entries()) {
      if (v === 1) return k
    }
  }
  console.log(func([1, 1, 1, 2]))
  console.log(func([1, 1, 1, 1, 2, 2, 2, 3]))
  
</script>
