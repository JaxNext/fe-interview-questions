# Q

- 给定驼峰风格的字符串，返回中线风格的结果
- 结果全小写

# A



# 总结



<script>
  function func(str) {
    const reg = /[A-Z]/g
    return str.replace(reg, (match, index) => {
      const char = match.toLowerCase()
      if (index === 0) {
        return char
      }
      return `-${char}`
    })
  }
  console.log(func('CamelCase'))
</script>
