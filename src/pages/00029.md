# Q

- 给定一个数组，保持原数组不变，去掉最小值；
- 如果有多个最小值，去掉 index 最小的那个
- 顺序不变

# A



# 总结



<script>
  function func(arr) {
    let min = arr[0]
    let index = 0
    for (let i = 1; i < arr.length; i++) {
      const x = arr[i]
      if (x < min) {
        min = x
        index = i
      }
    }
    arr.splice(index, 1)
    return arr
  }
  console.log(func([1, 2, 0, 3, 4, 5]))
  
</script>
