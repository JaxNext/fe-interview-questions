# Q

- 给定一个数字数组
- 将其中的奇数元素升序排序，保留偶数不动
- 若为空数组，则返回它

# A



# 总结



<script>
  function func(arr) {
    if (!arr?.length) return []
    const list = []
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] % 2 === 1) {
        list.push(arr[i])
      }
    }
    list.sort((a, b) => a - b)
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] % 2 === 1) {
        arr[i] = list.shift()
      }
    }
    return arr
  }
  console.log(func([3, 1, 2, 4, 7, 5, 6, 8, 9]))
  console.log(func([5, 3, 2, 8, 1, 4]))
</script>
