# Q

- 给定一个数组，元素为颜文字，求其中笑脸的个数。
- 笑脸必须有眼睛（: 或 ;）
- 笑脸必须有嘴（) 或 D）
- 鼻子（- 或 ~）非必须

# A



# 总结



<script>
  function func(str) {
    if (!str) return 0
    const reg = /[:;][-~]?[)D]/g
    const result = str.match(reg)
    return result?.length || 0
  }

  console.log(func('hello world'))
  console.log(func('hello world;)'))
  console.log(func('hello world;~'))
  console.log(func('hello world;~'))
  console.log(func('hello world;~D'))
  console.log(func('hello world;~D)'))
  console.log(func('hello world;~D)'))
  console.log(func('hello world;~D)'))
  console.log(func('hello world;~D)'))
  
</script>
