# 总结



<script>
  function func(obj) {
    let { type, value, children } = obj
    children = children || []
    console.log(type, value)
    if (!children.length) return
    const left = children.find(x => x.type === 'left')
    if(left) func(left)
    const right = children.find(x => x.type === 'right')
    if(right) func(right)
  }
  func({
    type: 'root',
    value: 0,
    children: [
      {
        type: 'left',
        value: 1,
        children: [
          {
            type: 'left',
            value: 2,
            children: []
          },
          {
            type: 'right',
            value: 2,
            children: []
          },
        ]
      },
      {
        type: 'right',
        value: 1,
        children: []
      }
    ]
  })
</script>
