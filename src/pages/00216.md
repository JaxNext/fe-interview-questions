<script>
  function func(arr) {
    if (arr.length <= 1) return arr
    const anchor = arr[0]
    const left = [], right = []
    for (let i = 1; i < arr.length; i++) {
      const x = arr[i]
      if (x < anchor) {
        left.push(x)
      } else {
        right.push(x)
      }
    }
    return [...func(left), anchor, ...func(right)]
  }
  console.log(func([3, 6, 8, 10, 1, 2, 1]))
</script>
