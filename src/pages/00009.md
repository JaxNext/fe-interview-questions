# 总结

- 除了用 map 计数，还可以用摩尔投票法
- 投票：假定第一个元素是目标，它每出现一次，票数加一，其他元素出现一次，票数减一。每当票数减到零，就换当次的元素作为假定目标

<script>
  // function func(arr) {
  //   const map = new Map()

  //   for(let x of arr) {

  function func(arr) {
    let count = 0
    let res = null
    for (let x of arr) {
      if (count === 0) {
        res = x
        count ++
      } else if (x === res) {
        count ++
      } else {
        count --
      }
    }
    return res
  }
  console.log(func([1, 3, 8, 2, 3, 1, 3, 3, 3]))
  console.log(func([5, 5, 5, 1, 2, 5, 5]))
  console.log(func([9, 9, 9, 9, 8, 9, 8, 8]))
</script>
